Store'Customers' table SQL
CREATE TABLE `store`.`customers` ( `Customer_ID` INT NOT NULL AUTO_INCREMENT , `First_Name` VARCHAR(255) NOT NULL , `Last_Name` VARCHAR(255) NOT NULL , `Email` VARCHAR(255) NOT NULL , `Full_Address` VARCHAR(255) NOT NULL , `Phone` VARCHAR(200) NOT NULL , `Password` VARCHAR(200) NOT NULL , `Credit_Card` VARCHAR(100) NOT NULL , `CardExpMo` SMALLINT NOT NULL , `CardExpYr` SMALLINT NOT NULL , `Date_Registered` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`Customer_ID`)) ENGINE = InnoDB;

Store'Suppliers' table SQL
CREATE TABLE `store`.`suppliers` ( `Supplier_ID` INT NOT NULL AUTO_INCREMENT , `Company_Name` VARCHAR(255) NOT NULL , `Contact_Name` VARCHAR(255) NOT NULL , `Company_Address` VARCHAR(255) NULL DEFAULT NULL , `Company_Email` VARCHAR(255) NOT NULL , `Company_Phone` VARCHAR(100) NOT NULL , `Company_URL` TEXT NOT NULL , PRIMARY KEY (`Supplier_ID`)) ENGINE = InnoDB;
ALTER TABLE `suppliers` ADD `Date_Registered` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP AFTER `Company_URL`;

Store'Orders' table SQL
CREATE TABLE `store`.`orders` ( `Order_ID` INT NOT NULL AUTO_INCREMENT , `Customer_ID` INT NOT NULL , `Order_Number` VARCHAR(8) NOT NULL , `Order_Date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `Delivery_Date` TIMESTAMP NULL DEFAULT NULL , `Transact_Status` ENUM("Pending","Completed","Cancelled") NOT NULL , `Paid` BOOLEAN NOT NULL DEFAULT FALSE , `Payment_Date` TIMESTAMP NULL DEFAULT NULL , `Payment_ID` INT NOT NULL , PRIMARY KEY (`Order_ID`), UNIQUE (`Customer_ID`)) ENGINE = InnoDB;

Store'Payment' table SQL
CREATE TABLE `store`.`payment` ( `Payment_ID` INT NOT NULL AUTO_INCREMENT , `Payment_Type` VARCHAR(255) NOT NULL , `Allowed` BOOLEAN NOT NULL , PRIMARY KEY (`Payment_ID`)) ENGINE = InnoDB;

Store'Categories' table SQL
CREATE TABLE `store`.`category` ( `Category_ID` INT NOT NULL AUTO_INCREMENT , `Category_Name` VARCHAR(100) NOT NULL , `Description` VARCHAR(200) NOT NULL , `Picture` VARCHAR(255) NOT NULL , `Active` BOOLEAN NOT NULL DEFAULT TRUE , PRIMARY KEY (`Category_ID`)) ENGINE = InnoDB;
